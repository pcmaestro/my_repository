/* ORACLE - JOIN de las tablas EMPLOYEES, DEPARTMENTS, JOBS, LOCATIONS, COUNTRIES y REGIONS para crear una tabla en la que se muestran
los empleados, su nombre,  su cargo,  su departamento y su ubaci√≥n */

SELECT E.EMPLOYEE_ID, E.FIRST_NAME, E.LAST_NAME, E.JOB_ID, J.JOB_TITLE, D.DEPARTMENT_ID, D.DEPARTMENT_NAME, L.CITY, C.COUNTRY_NAME, R.REGION_NAME 
FROM HR.EMPLOYEES E 
JOIN HR.DEPARTMENTS D 
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID 
JOIN HR.JOBS J 
ON E.JOB_ID = J.JOB_ID 
JOIN HR.LOCATIONS L 
ON D.LOCATION_ID = L.LOCATION_ID 
JOIN HR.COUNTRIES C 
ON C.COUNTRY_ID = L.COUNTRY_ID 
JOIN HR.REGIONS R 
ON C.REGION_ID = R.REGION_ID
